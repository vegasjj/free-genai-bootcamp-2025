        cursor.execute('SELECT id FROM groups WHERE id = ?', (id,))
        group = cursor.fetchone()
        if not group:
            return jsonify({"error": "Group not found"}), 404

        # Fetch all words for the group
        cursor.execute('''
            SELECT 
                w.id,
                w.kanji,
                w.romaji,
                w.english
            FROM words w
            JOIN group_words gw ON gw.word_id = w.id

            ORDER BY w.kanji
        ''', (id,))

                "romaji": word["romaji"],
